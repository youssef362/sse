const express = require("express"); const { createProxyMiddleware } = require("http-proxy-middleware"); const cors = require("cors"); const app = express(); app.use(cors()); // STREAMING + WEBSOCKET PROXY FOR UNITH app.use( "/avatar", createProxyMiddleware({ target: "https://stream.unith.ai", changeOrigin: true, ws: true, // enable websocket secure: true, pathRewrite: { "^/avatar": "", }, onProxyReq(proxyReq, req, res) { proxyReq.removeHeader("origin"); // prevent CORS issues }, onError(err, req, res) { console.error("Proxy error:", err); res.status(500).send("Proxy failed."); }, }) ); // Health check app.get("/health", (req, res) => res.send("OK")); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log("FULL STREAMING PROXY RUNNING on port " + PORT); });
